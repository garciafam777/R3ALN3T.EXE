void ATerminalSystem::BeginPlay()

    Super::BeginPlay();

void ATerminalSystem::InitializeTerminal()
{
    CurrentText = "";
    DisplayText("R3ALN3T.EXE v2.0.0\n> SYSTEMS NOMINAL.", false);
}

void ATerminalSystem::InitializeCommandResponses()
{
    CommandResponses.Add("help", "Available commands:\n- help: Show this message.\n- status: Check system status.\n- scan: Scan for devices.\n- debug: Toggle debug mode.\n- clear: Clear the terminal.\n- override: Skip cutscene (cheat).\n- secret: Unlock hidden lore.");
    CommandResponses.Add("status", "> N3T LINK: UNSTABLE\n> MEMORY: 7-G CORRUPTED\n> AI CORE: ONLINE\n> PET DEVICE: OFFLINE");
    CommandResponses.Add("scan", "SCANNING...\n> DEVICE FOUND: PET (ID: #XJ-482)\n> DEVICE FOUND: TERMINAL (ID: #XK-911)\n> WARNING: UNKNOWN SIGNAL DETECTED IN SECTOR 7-G.");
    CommandResponses.Add("debug", bDebugMode ? "DEBUG MODE: DISABLED" : "DEBUG MODE: ENABLED");
    CommandResponses.Add("clear", "");
    CommandResponses.Add("override", "OVERRIDE ACCEPTED.\n> SKIPPING CUTSCENE...");
    CommandResponses.Add("secret", "ACCESSING HIDDEN LORE...\n> FILE: //N3T/ARCHIVES/INCIDENT_3026\n> CONTENT: The bugmatter was a test. We failed.");
}

void ATerminalSystem::ProcessCommand(const FString& Command)
{
    if (Command.IsEmpty()) return;

    // Add the command to the terminal
    AddLine("> " + Command, false);

    // Process the command
    FString LowerCommand = Command.ToLower();
    if (CommandResponses.Contains(LowerCommand))
    {
        FString Response = CommandResponses[LowerCommand];
        if (LowerCommand == "clear")
        {
            ClearTerminal();
        }
        else if (LowerCommand == "debug")
        {
            bDebugMode = !bDebugMode;
            CommandResponses["debug"] = bDebugMode ? "DEBUG MODE: ENABLED" : "DEBUG MODE: DISABLED";
            DisplayText(Response, false);
        }
        else if (LowerCommand == "override")
        {
            DisplayText(Response, false);
            // Trigger cutscene skip (via Blueprint or Gameplay Static)
            UGameplayStatics::GetPlayerController(this, 0)->SetPause(false);
        }
        else
        {
            DisplayText(Response, false);
        }
    }
    else
    {
        DisplayText("ERROR: UNKNOWN COMMAND \"" + Command + "\"", true);
    }
}

void ATerminalSystem::DisplayText(const FString& Text, bool bIsError)
{
    if (Text.IsEmpty()) return;

    // Split text into lines
    TArray<FString> Lines;
    Text.ParseIntoArray(Lines, TEXT("\n"));

    for (const FString& Line : Lines)
    {
        AddLine(Line, bIsError);
    }
}

void ATerminalSystem::AddLine(const FString& Line, bool bIsError)
{
    if (CurrentText.IsEmpty())
    {
        CurrentText = Line;
    }
    else
    {
        CurrentText += "\n" + Line;
    }

    // Trim to MaxLines
    TArray<FString> AllLines;
    CurrentText.ParseIntoArray(AllLines, TEXT("\n"));
    if (AllLines.Num() > MaxLines)
    {
        AllLines.RemoveAt(0, AllLines.Num() - MaxLines);
        CurrentText = FString::Join(AllLines, TEXT("\n"));
    }

    // Broadcast the updated text (for UI)
    OnTerminalTextUpdated.Broadcast(CurrentText);
}

void ATerminalSystem::ClearTerminal()
{
    CurrentText = "";
    OnTerminalTextUpdated.Broadcast(CurrentText);
}

void ATerminalSystem::TriggerGlitchEffect()
{
    // Broadcast glitch effect (for UI)
    OnTerminalGlitch.Broadcast();
}

#include "TerminalMemoryManager.h"

void UTerminalMemoryManager::InitializeMemory(int32 SizeX, int32 SizeY)
{
    MemoryGrid.SetNum(SizeX);
    for (auto& Row : MemoryGrid)
    {
        Row.SetNum(SizeY);
    }
}

void UTerminalMemoryManager::StoreCommand(const FString& Command, const FString& Response)
{
    CommandMap.Add(Command, Response);
    // Store in memory grid (example: hash-based indexing)
    int32 X = FMath::Abs(Command.GetHash()) % MemoryGrid.Num();
    int32 Y = FMath::Abs(Response.GetHash()) % MemoryGrid[X].Num();
    MemoryGrid[X][Y] = Command + "|" + Response;
}

FString UTerminalMemoryManager::RetrieveCommand(const FString& Command)
{
    if (CommandMap.Contains(Command))
    {
        return CommandMap[Command];
    }
    return "ERROR: COMMAND NOT FOUND";
}

void UTerminalMemoryManager::ClearMemory()
{
    MemoryGrid.Empty();
    CommandMap.Empty();
}

void AInteractableActor::BeginPlay()
{
    Super::BeginPlay();

    // Store original materials
    if (MeshComponent)
    {
        for (int32 i = 0; i < MeshComponent->GetNumMaterials(); ++i)
        {
            OriginalMaterials.Add(MeshComponent->GetMaterial(i));
        }
    }
}

void AInteractableActor::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);
}

void AInteractableActor::OnLookAt()
{
    if (!bIsHighlighted)
    {
        Highlight();
        bIsHighlighted = true;
    }
}

void AInteractableActor::OnInteract()
{
    // Override in child classes (e.g., PETDevice, Terminal)
    UE_LOG(LogTemp, Warning, TEXT("Interacted with %s"), *InteractableName);
}

void AInteractableActor::Highlight()
{
    if (MeshComponent && HighlightMaterial)
    {
        MeshComponent->SetMaterial(0, HighlightMaterial);
    }
}

void AInteractableActor::RestoreMaterials()
{
    if (MeshComponent)
    {
        for (int32 i = 0; i < OriginalMaterials.Num(); ++i)
        {
            MeshComponent->SetMaterial(i, OriginalMaterials[i]);
        }
    }
    bIsHighlighted = false;
}

#include "InteractableActor.h"
#include "GameFramework/PlayerController.h"
#include "Camera/CameraComponent.h"

}

void UPlayerInteractComponent::BeginPlay()
{
    Super::BeginPlay();
}

void UPlayerInteractComponent::TickComponent(float DeltaTime, ELevelTick TickType, FActorComponentTickFunction* ThisTickFunction)
{
    Super::TickComponent(DeltaTime, TickType, ThisTickFunction);
    FindInteractable();
}

void UPlayerInteractComponent::Interact()
{
    if (CurrentInteractable)
    {
        CurrentInteractable->OnInteract();
    }
}

void UPlayerInteractComponent::FindInteractable()
{
    APlayerController* PlayerController = GetOwner<APlayerController>();
    if (!PlayerController) return;

    APawn* PlayerPawn = PlayerController->GetPawn();
    if (!PlayerPawn) return;

    UCameraComponent* PlayerCamera = PlayerPawn->FindComponentByClass<UCameraComponent>();
    if (!PlayerCamera) return;

    // Perform a line trace to find interactables
    FVector Start = PlayerCamera->GetComponentLocation();
    FVector End = Start + (PlayerCamera->GetForwardVector() * InteractDistance);

    FHitResult HitResult;
    FCollisionQueryParams QueryParams;
    QueryParams.AddIgnoredActor(PlayerPawn);

    if (GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, QueryParams))
    {
        AInteractableActor* Interactable = Cast<AInteractableActor>(HitResult.GetActor());
        if (Interactable)
        {
            if (Interactable != CurrentInteractable)
            {
                ClearInteractable();
                CurrentInteractable = Interactable;
                CurrentInteractable->OnLookAt();
            }
        }
        else
        {
            ClearInteractable();
        }
    }
    else
    {
        ClearInteractable();
    }
}

void UPlayerInteractComponent::ClearInteractable()
{
    if (CurrentInteractable)
    {
        CurrentInteractable->RestoreMaterials();
        CurrentInteractable = nullptr;
    }
}

{
    bHackingSuccess = bSuccess;
    UE_LOG(LogTemp, Warning, TEXT("Hacking game completed with %s"), bSuccess ? TEXT("success") : TEXT("failure"));

    if (bSuccess)
    {
        // Unlock lore or trigger a positive event
        ATerminalSystem* TerminalSystem = FindObjectOfClass<ATerminalSystem>(GetWorld());
        if (TerminalSystem)
        {
            TerminalSystem->DisplayText("> FILE UNLOCKED: //N3T/ARCHIVES/INCIDENT_3026\n> CONTENT: The bugmatter was a test. We failed.", false);
        }
    }
    else
    {
        // Trigger a negative event (e.g., security alert)
        ATerminalSystem* TerminalSystem = FindObjectOfClass<ATerminalSystem>(GetWorld());
        if (TerminalSystem)
        {
            TerminalSystem->DisplayText("> ACCESS DENIED. ALERT TRIGGERED.", true);
        }
    }
}

void AHackingGame::SubmitPassword(const FString& Password)
{
    if (!bGameActive) return;

    CurrentPassword = Password;
    RemainingAttempts--;

    if (Password == CorrectPassword)
    {
        // Success
        bGameActive = false;
        OnHackingGameCompleted.Broadcast(true);
    }
    else if (RemainingAttempts <= 0)
    {
        // Failure
        bGameActive = false;
        OnHackingGameCompleted.Broadcast(false);
    }
    else
    {
        // Wrong password, try again
        UE_LOG(LogTemp, Warning, TEXT("Wrong password. Attempts left: %d"), RemainingAttempts);
    }
}

AMemoryReconstructionGame::AMemoryReconstructionGame()
{
    PrimaryActorTick.bCanEverTick = false;
}

void AMemoryReconstructionGame::BeginPlay()
{
    Super::BeginPlay();
}

void AMemoryReconstructionGame::StartGame(const TArray<FMemoryFragment>& Fragments, const FString& Reward)
{
    OriginalFragments = Fragments;
    LoreReward = Reward;
    CurrentFragments = OriginalFragments;
    ShuffleFragments();
    bGameActive = true;
    OnMemoryGameCompleted.Clear();
}

void AMemoryReconstructionGame::ShuffleFragments()
{
    // Fisher-Yates shuffle
    for (int32 i = CurrentFragments.Num() - 1; i > 0; --i)
    {
        int32 j = FMath::RandRange(0, i);
        if (i != j)
        {
            CurrentFragments.Swap(i, j);
        }
    }
}

void AMemoryReconstructionGame::SubmitReconstruction()
{
    if (!bGameActive) return;

    if (IsReconstructionCorrect())
    {
        bGameActive = false;
        OnMemoryGameCompleted.Broadcast(true, LoreReward);
    }
    else
    {
        OnMemoryGameCompleted.Broadcast(false, "Reconstruction failed. Try again.");
    }
}

bool AMemoryReconstructionGame::IsReconstructionCorrect()
{
    for (int32 i = 0; i < CurrentFragments.Num(); ++i)
    {
        if (CurrentFragments[i].CorrectPosition != i)
        {
            return false;
        }
    }
    return true;
}

void UMemoryReconstructionWidget::OnFragmentDragged(int32 FragmentIndex, int32 NewPosition)
{
    if (FragmentIndex < 0 || FragmentIndex >= CurrentFragments.Num() || NewPosition < 0 || NewPosition >= CurrentFragments.Num())
    {
        return;
    }

    // Swap fragments
    FMemoryFragment Temp = CurrentFragments[FragmentIndex];
    CurrentFragments[FragmentIndex] = CurrentFragments[NewPosition];

    if (bGameActive)
    {
        UpdateGame(DeltaTime);
    }

void ABugmatterContainmentGame::StartGame(float TimeLimit, int32 RequiredSuccesses)
{
    this->TimeLimit = TimeLimit;
    this->TimeRemaining = TimeLimit;
    this->RequiredSuccesses = RequiredSuccesses;
    this->CurrentSuccesses = 0;
    this->CurrentProgress = 0.0f;
    bGameActive = true;
    OnBugmatterGameCompleted.Clear();
}

void ABugmatterContainmentGame::OnKeyPressed(FKey Key)
{
    if (!bGameActive) return;

    if (ValidKeys.Contains(Key))
    {
        CurrentSuccesses++;
        CurrentProgress = static_cast<float>(CurrentSuccesses) / RequiredSuccesses;

        if (CurrentSuccesses >= RequiredSuccesses)
        {
            // Success
            bGameActive = false;
            OnBugmatterGameCompleted.Broadcast(true);
        }
    }
}

void ABugmatterContainmentGame::UpdateGame(float DeltaTime)
{
    TimeRemaining -= DeltaTime;
    if (TimeRemaining <= 0.0f)
    {
        // Failure
        bGameActive = false;
        OnBugmatterGameCompleted.Broadcast(false);
    }
}

#include "NaviDialogueSystem.h"

ANaviDialogueSystem::ANaviDialogueSystem()
{
    PrimaryActorTick.bCanEverTick = false;
}

void ANaviDialogueSystem::BeginPlay()
{
    Super::BeginPlay();
}

void ANaviDialogueSystem::StartDialogue(const FDialogueNode& RootNode)
{
    CurrentNode = RootNode;
    bDialogueActive = true;
    DialogueHistory.Empty();
    DialogueHistory.Add(CurrentNode);
    OnDialogueUpdated.Broadcast(CurrentNode);
}

void ANaviDialogueSystem::SelectOption(int32 OptionIndex)
{
    if (!bDialogueActive || OptionIndex < 0 || OptionIndex >= CurrentNode.Options.Num())
    {
        return;
    }

    const FDialogueOption& SelectedOption = CurrentNode.Options[OptionIndex];
    if (SelectedOption.NextOptions.Num() > 0)
    {
        // Move to the next node
        FDialogueNode NextNode;
        NextNode.SpeakerName = CurrentNode.SpeakerName;
        NextNode.DialogueText = SelectedOption.ResponseText;
        NextNode.Options = SelectedOption.NextOptions;

        DialogueHistory.Add(NextNode);
        CurrentNode = NextNode;
        OnDialogueUpdated.Broadcast(CurrentNode);
    }
    else
    {
        // End the dialogue
        bDialogueActive = false;
        OnDialogueUpdated.Broadcast(CurrentNode);
    }
}

#include "InventorySystem.h"

bool UInventorySystem::AddItem(const FInventoryItem& Item)
{
    if (Item.ItemID.IsEmpty()) return false;

    if (ItemMap.Contains(Item.ItemID))
    {
        // Item already exists; increment quantity if needed
        return false;
    }

    Inventory.Add(Item);
    ItemMap.Add(Item.ItemID, Item);
    OnInventoryUpdated.Broadcast();
    return true;
}

bool UInventorySystem::RemoveItem(const FString& ItemID)
{
    if (!ItemMap.Contains(ItemID)) return false;

    for (int32 i = 0; i < Inventory.Num(); ++i)
    {
        if (Inventory[i].ItemID == ItemID)
        {
            Inventory.RemoveAt(i);
            ItemMap.Remove(ItemID);
            OnInventoryUpdated.Broadcast();
            return true;
        }
    }
    return false;
}

bool UInventorySystem::UseItem(const FString& ItemID)
{
    if (!ItemMap.Contains(ItemID)) return false;

    FInventoryItem Item = ItemMap[ItemID];
    if (!Item.bIsUsable) return false;

    // Trigger the item's use effect
    if (!Item.UseCommand.IsEmpty())
    {
        // Example: Send the UseCommand to the TerminalSystem
        ATerminalSystem* TerminalSystem = FindObjectOfClass<ATerminalSystem>(GetOuter()->GetWorld());
        if (TerminalSystem)
        {
            TerminalSystem->ProcessCommand(Item.UseCommand);
        }
    }

    return true;
}

FInventoryItem UInventorySystem::GetItem(const FString& ItemID) const
{
    if (ItemMap.Contains(ItemID))
    {
        return ItemMap[ItemID];
    }
    return FInventoryItem();
}

bool UInventorySystem::HasItem(const FString& ItemID) const
{
    return ItemMap.Contains(ItemID);
}

void UInventoryWidget::NativeConstruct()
{
    Super::NativeConstruct();
}

void UInventoryWidget::InitializeWidget(UInventorySystem* System)
{
    InventorySystem = System;
    if (InventorySystem)
    {
        InventorySystem->OnInventoryUpdated.AddDynamic(this, &UInventoryWidget::OnInventoryUpdated);
        UpdateInventoryUI();
    }
}

void UInventoryWidget::OnInventoryUpdated()
{
    UpdateInventoryUI();
}

void UInventoryWidget::UpdateInventoryUI()
{
    if (!InventorySystem || !InventoryPanel) return;

    InventoryPanel->ClearChildren();

    TArray<FInventoryItem> Inventory = InventorySystem->GetInventory();
    for (const FInventoryItem& Item : Inventory)
    {
        UBorder* ItemBorder = WidgetTree::GetWidgetTree(this)->ConstructWidget<UBorder>(UBorder::StaticClass());
        ItemBorder->SetPadding(FMargin(5, 5, 5, 5));
        ItemBorder->SetBrushColor(FLinearColor::Black);

        if (Item.ItemIcon)
        {
            UImage* ItemImage = WidgetTree::GetWidgetTree(this)->ConstructWidget<UImage>(UImage::StaticClass());
            ItemImage->SetBrushFromTexture(Item.ItemIcon);
            ItemImage->SetRenderScale(FVector2D(0.8f, 0.8f));
            ItemBorder->AddChildToBorder(ItemImage);
        }

        UTextBlock* ItemNameText = WidgetTree::GetWidgetTree(this)->ConstructWidget<UTextBlock>(UTextBlock::StaticClass());
        ItemNameText->SetText(FText::FromString(Item.ItemName));
        ItemNameText->SetFont(FSlateFontInfo("ShareTechMono", 10));
        ItemNameText->SetColorAndOpacity(FLinearColor::Green);

        UVerticalBox* ItemBox = WidgetTree::GetWidgetTree(this)->ConstructWidget<UVerticalBox>(UVerticalBox::StaticClass());
        ItemBox->AddChildToVerticalBox(ItemImage);
        ItemBox->AddChildToVerticalBox(ItemNameText);
        ItemBorder->AddChildToBorder(ItemBox);

        // Store the ItemID as a tag
        ItemBorder->AddTag(FName(*Item.ItemID));

        // Add click handler
        UButton* ItemButton = WidgetTree::GetWidgetTree(this)->ConstructWidget<UButton>(UButton::StaticClass());
        ItemButton->AddChildToBorder(ItemBorder);
        ItemButton->OnClicked.AddDynamic(this, &UInventoryWidget::OnItemSelected, Item.ItemID);

        InventoryPanel->AddChildToPanel(ItemButton);
    }
}

void UInventoryWidget::OnItemSelected(const FString& ItemID)
{
    if (InventorySystem)
    {
        FInventoryItem Item = InventorySystem->GetItem(ItemID);
        if (ItemDescriptionText)
        {
            ItemDescriptionText->SetText(FText::FromString(Item.ItemDescription));
        }

        // Use the item
        InventorySystem->UseItem(ItemID);
    }
}

🛠 Tips for Designers


Feedback:

Provide visual/audio feedback for all interactions (e.g., highlight hotspots, play sounds).
Use tooltips or descriptions to guide players.


Pacing:

Keep interactions optional to avoid breaking cutscene flow.
Use time limits for mini-games to maintain tension.


Rewards:

Reward exploration with lore, secrets, or alternate paths.
Avoid punishing players for missing interactions.


Aesthetic:

Match the cyberpunk UI style (neon colors, glitch effects, monospace fonts).
Use sound design to enhance immersion (e.g., terminal beeps, glitch static).
