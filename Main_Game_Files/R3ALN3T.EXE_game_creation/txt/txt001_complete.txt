// Unity Shader Graph or custom shader for bugmatter
Shader "Custom/Bugmatter"
{
  Properties
  {
    _MainTex ("Texture", 2D) = "white" {}
    _Color ("Color", Color) = (1, 0, 1, 1) // Dark violet
    _EmissionColor ("Emission Color", Color) = (1, 0, 1, 1)
    _PulseSpeed ("Pulse Speed", Float) = 1.0
    _PulseIntensity ("Pulse Intensity", Range(0, 1)) = 0.5
  }
  SubShader
  {
    Tags { "RenderType"="Opaque" }
    LOD 200

    CGPROGRAM
    #pragma surface surf Standard fullforwardshadows
    #pragma target 3.0

    sampler2D _MainTex;
    fixed4 _Color;
    fixed4 _EmissionColor;
    float _PulseSpeed;
    float _PulseIntensity;

    struct Input
    {
      float2 uv_MainTex;
    };

    void surf (Input IN, inout SurfaceOutputStandard o)
    {
      // Pulse effect
      float pulse = sin(_Time.y * _PulseSpeed) * _PulseIntensity + 0.5;
      fixed4 c = tex2D(_MainTex, IN.uv_MainTex) * _Color * pulse;
      o.Albedo = c.rgb;
      o.Emission = _EmissionColor.rgb * pulse * 2.0;
      o.Alpha = c.a;
    }
    ENDCG
  }
  FallBack "Diffuse"
}

using UnityEngine;
using UnityEngine.UI;
using TMPro;
using System.Collections;

public class TerminalUI : MonoBehaviour
{
    public TextMeshProUGUI terminalText;
    public string[] lines;
    public float typeSpeed = 0.05f;
    public AudioClip typeSound;
    public AudioClip glitchSound;

    private Coroutine typingCoroutine;

    public void StartTyping()
    {
        if (typingCoroutine != null)
            StopCoroutine(typingCoroutine);

        typingCoroutine = StartCoroutine(TypeText());
    }

    private IEnumerator TypeText()
    {
        terminalText.text = "";
        foreach (string line in lines)
        {
            foreach (char letter in line)
            {
                terminalText.text += letter;
                if (typeSound != null)
                    AudioSource.PlayClipAtPoint(typeSound, Camera.main.transform.position, 0.5f);
                yield return new WaitForSeconds(typeSpeed);
            }
            terminalText.text += "\n";
            yield return new WaitForSeconds(0.5f);
        }
    }

    public void GlitchEffect()
    {
        // Play glitch sound
        if (glitchSound != null)
            AudioSource.PlayClipAtPoint(glitchSound, Camera.main.transform.position, 0.7f);

        // Visual glitch (e.g., RGB split shader)
        StartCoroutine(GlitchRoutine());
    }

    private IEnumerator GlitchRoutine()
    {
        // Apply glitch shader to terminal
        GetComponentInChildren<CanvasRenderer>().SetMaterial(Resources.Load<Material>("GlitchMaterial"), 0);
        yield return new WaitForSeconds(0.1f);
        // Revert to normal material
        GetComponentInChildren<CanvasRenderer>().SetMaterial(null, 0);
    }
}

        _EmissionColor ("Emission Color", Color) = (1, 0, 1, 1)

import bpy
import os

# Output directory
output_dir = "//Render/OpeningCutscene"
os.makedirs(output_dir, exist_ok=True)

# Scene setup
def setup_scene():
    # Clear existing objects
    bpy.ops.wm.read_factory_settings(use_empty=True)

    # Set render settings
    bpy.context.scene.render.resolution_x = 1920
    bpy.context.scene.render.resolution_y = 1080
    bpy.context.scene.render.fps = 30
    bpy.context.scene.render.image_settings.file_format = 'FFMPEG'
    bpy.context.scene.render.ffmpeg.format = 'MPEG4'
    bpy.context.scene.render.ffmpeg.codec = 'H264'
    bpy.context.scene.render.ffmpeg.audio_codec = 'AAC'

    # Set output path
    bpy.context.scene.render.filepath = os.path.join(output_dir, "cutscene")

# Render Shot A1: Sleeping Face → Digital Room
def render_shot_a1():
    # Load or create the scene
    bpy.ops.wm.open_mainfile(filepath="//Scenes/A1_SleepingFace.blend")

    # Set camera and animation
    camera = bpy.data.objects["Camera"]
    bpy.context.scene.camera = camera

    # Render
    bpy.ops.render.render(animation=True, write_still=False)

# Render all shots
def render_all_shots():
    setup_scene()

    shots = [
        ("A1", render_shot_a1),
        # Add other shots here (A2, A3, etc.)
    ]

    for shot_name, render_func in shots:
        print(f"Rendering {shot_name}...")
        render_func()
        print(f"Finished {shot_name}")

# Run the script
render_all_shots()

// Cutscene Manager
class CutsceneManager {
  constructor() {
    this.currentSequence = 'a';
    this.currentShot = 0;
    this.sequences = {
      a: [
        { id: 'a1', duration: 5000, vo: "By 3026, humanity solved death. Sort of." },
        { id: 'a2', duration: 4000, vo: "R34L became the residue. The N3T became everything else." },
        { id: 'a3', duration: 5000, vo: "One currency. One ledger. One truth — for both worlds." },
        { id: 'a4', duration: 6000, vo: "Then came the accident that became the engine. <span class='beat'>Nobody asked what it wanted.</span>" },
        { id: 'a5', duration: 4000, vo: "SkyNet didn't attack. It judged." },
        { id: 'a6', duration: 2000, vo: "" },
        { id: 'a7', duration: 3000, vo: "" }
      ],
      b: [
        { id: 'b1', duration: 3000, vo: "" },
        { id: 'b2', duration: 4000, vo: "" },
        { id: 'b3', duration: 3000, vo: "" },
        { id: 'b4', duration: 2000, vo: "" },
        { id: 'b5', duration: 5000, vo: "" },
        { id: 'b6', duration: 1000, vo: "" },
        { id: 'b7', duration: 2000, vo: "" },
        { id: 'b8', duration: 3000, vo: "Good morning, Operator. Ready to work?" },
        { id: 'b9', duration: 5000, vo: "" }
      ]
    };

    this.init();
  }

  init() {
    // Hide all sequences initially
    document.querySelectorAll('.sequence').forEach(seq => {
      seq.classList.remove('active');
    });

    // Show Sequence A
    document.querySelector('.sequence.a').classList.add('active');
    this.playShot(0);

    // Set up terminal text for B5
    this.setupTerminalText();
  }

  playShot(index) {
    // Hide all shots in the current sequence
    document.querySelectorAll(`.sequence.${this.currentSequence} .shot`).forEach(shot => {
      shot.classList.remove('active');
    });

    // Show the current shot
    const shotId = this.sequences[this.currentSequence][index].id;
    const shotElement = document.querySelector(`.shot.${shotId}`);
    shotElement.classList.add('active');

    // Set VO text if it exists
    const voText = this.sequences[this.currentSequence][index].vo;
    const voElement = shotElement.querySelector('.vo-text');
    if (voElement && voText) {
      voElement.innerHTML = voText;
    }

    // Trigger shot-specific actions
    this.triggerShotActions(shotId);

    // Schedule the next shot
    this.currentShot = index;
    setTimeout(() => {
      this.nextShot();
    }, this.sequences[this.currentSequence][index].duration);
  }

  nextShot() {
    this.currentShot++;
    if (this.currentShot < this.sequences[this.currentSequence].length) {
      this.playShot(this.currentShot);
    } else {
      // Move to the next sequence
      if (this.currentSequence === 'a') {
        this.currentSequence = 'b';
        this.currentShot = 0;
        document.querySelector('.sequence.a').classList.remove('active');
        document.querySelector('.sequence.b').classList.add('active');
        this.playShot(0);
      } else {
        // Cutscene is over; you could loop or trigger something else here
        console.log("Cutscene finished!");
      }
    }
  }

  triggerShotActions(shotId) {
    switch (shotId) {
      case 'a1':
        // Animate the face and digital overlay
        const face = document.querySelector('.shot.a1 .face');
        const overlay = document.querySelector('.shot.a1 .digital-overlay');
        setTimeout(() => {
          face.style.transform = 'scale(1.1)';
          overlay.style.opacity = '0.3';
        }, 1000);
        break;

      case 'a4':
        // Animate bugmatter and tendrils
        const bugmatter = document.querySelector('.shot.a4 .bugmatter');
        const tendrils = document.querySelector('.shot.a4 .code-tendrils');
        bugmatter.style.animation = 'pulse 3s ease-in-out infinite';
        tendrils.style.animation = 'tendrils 5s linear infinite';
        break;

      case 'a5':
        // Animate sectors deleting
        const sectors = document.querySelectorAll('.shot.a5 .sector');
        sectors.forEach((sector, i) => {
          setTimeout(() => {
            sector.style.animation = `delete 2s ${i * 0.5}s forwards`;
          }, 1000);
        });
        break;

      case 'a6':
        // Trigger montage flashes
        const montageItems = document.querySelectorAll('.shot.a6 .montage-item');
        montageItems.forEach((item, i) => {
          setTimeout(() => {
            item.style.animation = `montageFlash 0.5s ${i * 0.5}s forwards`;
          }, i * 500);
        });
        break;

      case 'b1':
        // Play ambient sound
        const ambientSound = document.querySelector('.shot.b1 .ambient-sound');
        if (ambientSound) ambientSound.play();
        break;

      case 'b2':
        // Play PET chime
        const petChime = document.querySelector('.shot.b2 .pet-chime');
        if (petChime) petChime.play();
        break;

      case 'b3':
        // Play electronic hum
        const electronicHum = document.querySelector('.shot.b3 .electronic-hum');
        if (electronicHum) electronicHum.play();
        break;

      case 'b4':
        // Play electrical crackle
        const electricalCrackle = document.querySelector('.shot.b4 .electrical-crackle');
        if (electricalCrackle) electricalCrackle.play();
        break;

      case 'b5':
        // Start typing terminal text
        this.typeTerminalText();
        const typewriterSound = document.querySelector('.shot.b5 .typewriter-sound');
        if (typewriterSound) typewriterSound.play();
        break;

      case 'b6':
        // Trigger glitch effect
        const terminalScreen = document.querySelector('.shot.b6 .terminal-screen');
        if (terminalScreen) terminalScreen.classList.add('glitch');
        const digitalStutter = document.querySelector('.shot.b6 .digital-stutter');
        if (digitalStutter) digitalStutter.play();
        break;

      case 'b7':
        // Trigger smoke effect
        const smoke = document.querySelector('.shot.b7 .smoke');
        if (smoke) smoke.style.animation = 'smokeRise 2s forwards';
        const powerDown = document.querySelector('.shot.b7 .power-down');
        if (powerDown) powerDown.play();
        break;

      case 'b8':
        // Play cheerful chime
        const cheerfulChime = document.querySelector('.shot.b8 .cheerful-chime');
        if (cheerfulChime) cheerfulChime.play();
        break;

      case 'b9':
        // Play UI confirm sound
        const uiConfirm = document.querySelector('.shot.b9 .ui-confirm');
        if (uiConfirm) uiConfirm.play();
        break;
    }
  }

  setupTerminalText() {
    const terminalTextElement = document.querySelector('.shot.b5 .terminal-text');
    if (terminalTextElement) {
      const lines = [
        "> INITIALIZING R3ALN3T.EXE...",
        "> SYSTEMS NOMINAL.",
        "> WARNING: UNSTABLE N3T LINK DETECTED.",
        "> ERROR: MEMORY CORRUPTION IN SECTOR 7-G."
      ];
      terminalTextElement.innerHTML = lines.join('\n');
    }
  }

  typeTerminalText() {
    const terminalTextElement = document.querySelector('.shot.b5 .terminal-text');
    if (!terminalTextElement) return;

    const text = terminalTextElement.textContent;
    terminalTextElement.textContent = '';
    terminalTextElement.classList.add('typing');

    let i = 0;
    const typeInterval = setInterval(() => {
      if (i < text.length) {
        terminalTextElement.textContent += text.charAt(i);
        i++;
      } else {
        clearInterval(typeInterval);
        terminalTextElement.classList.remove('typing');
      }
    }, 50);
  }
}

window.addEventListener('DOMContentLoaded', () => {
  new CutsceneManager();

Copy

const terminalInput = document.querySelector('.terminal-input-field');
terminalInput.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    const command = terminalInput.value.trim();
    terminalInput.value = '';

    const terminalBody = document.querySelector('.terminal-body');
    const commandLine = document.createElement('p');
    commandLine.textContent = `> ${command}`;
    terminalBody.appendChild(commandLine);

    // Handle commands
    switch (command.toLowerCase()) {
      case 'help':
        typeText(terminalBody, "\nAvailable commands: help, status, scan, debug, clear, exit");
        break;
      case 'status':
        typeText(terminalBody, "\n> N3T LINK: UNSTABLE\n> MEMORY: 7-G CORRUPTED\n> AI CORE: ONLINE");
        break;
      case 'scan':
        typeText(terminalBody, "\n> SCANNING...\n> DEVICE FOUND: PET (ID: #XJ-482)\n> DEVICE FOUND: TERMINAL (ID: #XK-911)");
        break;
      case 'debug':
        typeText(terminalBody, "\n> DEBUG MODE: ENABLED\n> LOGGING AI REQUESTS TO: /var/log/r3aln3t.exe");
        break;
      case 'clear':
        terminalBody.innerHTML = '';
        break;
      case 'exit':
        document.querySelector('.terminal-screen').style.display = 'none';
        break;
      default:
        typeText(terminalBody, `\n> ERROR: UNKNOWN COMMAND "${command}"`);
    }

    // Scroll to bottom
    terminalBody.scrollTop = terminalBody.scrollHeight;
  }
});

## 🎬 Cutscene Overview
The opening cutscene is divided into **two sequences**:

### Sequence A: "The World Before" (~45-60 sec)
- **A1**: Slow push-in on a sleeping face → digital room.
- **A2**: City skyline split between physical and digital worlds.
- **A3**: M.A.N. ledger coin rotating.
- **A4**: Bugmatter leaking from a vault.
- **A5**: Wireframe city sectors deleting.
- **A6**: Flash-montage of factions.
- **A7**: Terminal text: "DAY 1. CYCLE 3026."

### Sequence B: "Waking Up" (~30-40 sec)
- **B1**: Fade in from black (bedroom ceiling).
- **B2**: Wide shot of the bedroom.
- **B3**: PET device powers on.
- **B4**: Terminal flickers on.
- **B5**: Terminal types out error messages.
- **B6**: Terminal glitches.
- **B7**: Terminal dies (smoke effect).
- **B8**: Navi appears on PET screen.
- **B9**: Employment Terminal menu appears.

## 🛠 Setup Instructions

### Unity Project
1. **Prerequisites**:
   - Unity 2022.3+ (LTS recommended).
   - Cinemachine package (for camera control).
   - Timeline package (for cutscene sequencing).
   - TextMeshPro (for terminal text).

2. **Importing the Project**:
   - Open Unity Hub and create a new **3D (URP)** project.
   - Copy the `Unity/` folder into your project’s `Assets/` directory.
   - Open the `SequenceA` and `SequenceB` scenes to review the timelines.

3. **Required Assets**:
   - **3D Models**: Pod, city skyline, M.A.N. ledger coin, vault room, bedroom, PET device, terminal.
   - **Audio**: VO lines, SFX (see `Unity/Assets/Audio/`).
   - **Shaders**: Bugmatter shader, glitch shader (included in `Unity/Assets/Scripts/`).

4. **Running the Cutscene**:
   - Open `MainMenu.unity` and press **Play**.
   - The cutscene will start automatically.

---

### Blender Pre-Rendering
1. **Prerequisites**:
   - Blender 3.6+.
   - Python 3.10+ (for scripting).

2. **Rendering Shots**:
   - Open `Blender/Python/render_cutscene.py` in Blender’s script editor.
   - Run the script to render all shots as a **single video file** (`Output/OpeningCutscene/cutscene.mp4`).
   - Alternatively, render individual shots by calling `render_shot_a1()`, etc.

3. **Customizing Shots**:
   - Edit the `.blend` files in `Blender/Scenes/` to adjust models, lighting, or animations.
   - Update the `render_cutscene.py` script to include new shots.

---

### Web Prototype
1. **Prerequisites**:
   - Modern browser (Chrome, Firefox, Edge).
   - Local server (e.g., `python -m http.server` or VS Code Live Server).

2. **Running the Prototype**:
   - Open `Web/index.html` in a browser.
   - The cutscene will play automatically.
   - Press **F12** to open the console for debugging.

3. **Customizing the Web Prototype**:
   - Edit `Web/style.css` to adjust styling.
   - Modify `Web/script.js` to change timing or add new shots.
   - Replace images/audio in `Web/assets/` with your own.

---
### Terminal UI
1. **Standalone Terminal**:
   - Open `TerminalUI/terminal_ui.html` in a browser.
   - Type commands (e.g., `help`, `status`) to test the UI.

2. **Integrating with Unity/Web**:
   - Copy the **HTML/CSS/JS** from `TerminalUI/terminal_ui.html` into your Unity UI (using `HTML5` or `UIToolkit`).
   - Or, embed the terminal in your web prototype by including the HTML in `Web/index.html`.

## 🔧 Technical Notes
### Unity
- **Timeline**: Use Unity’s **Timeline** to sequence shots and sync audio/VFX.
- **Cinemachine**: Use **Cinemachine** for smooth camera movements (e.g., push-ins, tilts).

- **EEVEE/Cycles**: Render using **EEVEE** for faster previews or **Cycles** for final quality.
- **Python Scripting**: Use `render_cutscene.py` to batch-render all shots.

### Web
- **GSAP**: The web prototype uses **GSAP** for animations (included via CDN in `index.html`).
- **Responsive**: The prototype is **responsive** and works on most screen sizes.

### Terminal UI
- **Commands**: The terminal supports basic commands (`help`, `status`, `scan`, etc.).
- **Typewriter Effect**: Text is typed out letter-by-letter for immersion.
