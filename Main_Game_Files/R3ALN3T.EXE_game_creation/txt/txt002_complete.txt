void ATerminalSystem::InitializeTerminal()
{
    CurrentText = "";
    DisplayText("R3ALN3T.EXE v2.0.0\n> SYSTEMS NOMINAL.", false);
}

void ATerminalSystem::InitializeCommandResponses()
{
    CommandResponses.Add("help", "Available commands:\n- help: Show this message.\n- status: Check system status.\n- scan: Scan for devices.\n- debug: Toggle debug mode.\n- clear: Clear the terminal.\n- override: Skip cutscene (cheat).\n- secret: Unlock hidden lore.");
    CommandResponses.Add("status", "> N3T LINK: UNSTABLE\n> MEMORY: 7-G CORRUPTED\n> AI CORE: ONLINE\n> PET DEVICE: OFFLINE");
    CommandResponses.Add("scan", "SCANNING...\n> DEVICE FOUND: PET (ID: #XJ-482)\n> DEVICE FOUND: TERMINAL (ID: #XK-911)\n> WARNING: UNKNOWN SIGNAL DETECTED IN SECTOR 7-G.");
    CommandResponses.Add("debug", bDebugMode ? "DEBUG MODE: DISABLED" : "DEBUG MODE: ENABLED");
    CommandResponses.Add("clear", "");
    CommandResponses.Add("override", "OVERRIDE ACCEPTED.\n> SKIPPING CUTSCENE...");
    CommandResponses.Add("secret", "ACCESSING HIDDEN LORE...\n> FILE: //N3T/ARCHIVES/INCIDENT_3026\n> CONTENT: The bugmatter was a test. We failed.");
}

void ATerminalSystem::ProcessCommand(const FString& Command)
{
    if (Command.IsEmpty()) return;

    // Add the command to the terminal
    AddLine("> " + Command, false);

    // Process the command
    FString LowerCommand = Command.ToLower();
    if (CommandResponses.Contains(LowerCommand))
    {
        FString Response = CommandResponses[LowerCommand];
        if (LowerCommand == "clear")
        {
            ClearTerminal();
        }
        else if (LowerCommand == "debug")
        {
            bDebugMode = !bDebugMode;
            CommandResponses["debug"] = bDebugMode ? "DEBUG MODE: ENABLED" : "DEBUG MODE: DISABLED";
            DisplayText(Response, false);
        }
        else if (LowerCommand == "override")
        {
            DisplayText(Response, false);
            // Trigger cutscene skip (via Blueprint or Gameplay Static)
            UGameplayStatics::GetPlayerController(this, 0)->SetPause(false);
        }
        else
        {
            DisplayText(Response, false);
        }
    }
    else
    {
        DisplayText("ERROR: UNKNOWN COMMAND \"" + Command + "\"", true);
    }
}

void ATerminalSystem::DisplayText(const FString& Text, bool bIsError)
{
    if (Text.IsEmpty()) return;

    // Split text into lines
    TArray<FString> Lines;
    Text.ParseIntoArray(Lines, TEXT("\n"));

    for (const FString& Line : Lines)
    {
        AddLine(Line, bIsError);
    }
}

void ATerminalSystem::AddLine(const FString& Line, bool bIsError)
{
    if (CurrentText.IsEmpty())
    {
        CurrentText = Line;
    }
    else
    {
        CurrentText += "\n" + Line;
    }

    // Trim to MaxLines
    TArray<FString> AllLines;
    CurrentText.ParseIntoArray(AllLines, TEXT("\n"));
    if (AllLines.Num() > MaxLines)
    {
        AllLines.RemoveAt(0, AllLines.Num() - MaxLines);
        CurrentText = FString::Join(AllLines, TEXT("\n"));
    }

    // Broadcast the updated text (for UI)
    OnTerminalTextUpdated.Broadcast(CurrentText);
}

void ATerminalSystem::ClearTerminal()
{
    CurrentText = "";
    OnTerminalTextUpdated.Broadcast(CurrentText);
}

void ATerminalSystem::TriggerGlitchEffect()
{
    // Broadcast glitch effect (for UI)
    OnTerminalGlitch.Broadcast();
}

#include "TerminalMemoryManager.h"

void UTerminalMemoryManager::InitializeMemory(int32 SizeX, int32 SizeY)
{
    MemoryGrid.SetNum(SizeX);
    for (auto& Row : MemoryGrid)
    {
        Row.SetNum(SizeY);
    }
}

void UTerminalMemoryManager::StoreCommand(const FString& Command, const FString& Response)
{
    CommandMap.Add(Command, Response);
    // Store in memory grid (example: hash-based indexing)
    int32 X = FMath::Abs(Command.GetHash()) % MemoryGrid.Num();
    int32 Y = FMath::Abs(Response.GetHash()) % MemoryGrid[X].Num();
    MemoryGrid[X][Y] = Command + "|" + Response;
}

FString UTerminalMemoryManager::RetrieveCommand(const FString& Command)
{
    if (CommandMap.Contains(Command))
    {
        return CommandMap[Command];
    }
    return "ERROR: COMMAND NOT FOUND";
}

void UTerminalMemoryManager::ClearMemory()
{
    MemoryGrid.Empty();
    CommandMap.Empty();
}

{
    Super::BeginPlay();

    // Store original materials
    if (MeshComponent)
    {
        for (int32 i = 0; i < MeshComponent->GetNumMaterials(); ++i)
        {
            OriginalMaterials.Add(MeshComponent->GetMaterial(i));
        }
    }
}

{
    Super::Tick(DeltaTime);
}

void AInteractableActor::OnLookAt()
{
    if (!bIsHighlighted)
    {
        Highlight();
        bIsHighlighted = true;
    }
}

void AInteractableActor::OnInteract()
{
    // Override in child classes (e.g., PETDevice, Terminal)
    UE_LOG(LogTemp, Warning, TEXT("Interacted with %s"), *InteractableName);
}

void AInteractableActor::Highlight()
{
    if (MeshComponent && HighlightMaterial)
    {
        MeshComponent->SetMaterial(0, HighlightMaterial);
    }
}

void AInteractableActor::RestoreMaterials()
{
    if (MeshComponent)
    {
        for (int32 i = 0; i < OriginalMaterials.Num(); ++i)
        {
            MeshComponent->SetMaterial(i, OriginalMaterials[i]);
        }
    }
    bIsHighlighted = false;
}

}

void UPlayerInteractComponent::BeginPlay()
{
    Super::BeginPlay();
}

void UPlayerInteractComponent::TickComponent(float DeltaTime, ELevelTick TickType, FActorComponentTickFunction* ThisTickFunction)
{
    Super::TickComponent(DeltaTime, TickType, ThisTickFunction);
    FindInteractable();
}

void UPlayerInteractComponent::Interact()
{
    if (CurrentInteractable)
    {
        CurrentInteractable->OnInteract();
    }
}

void UPlayerInteractComponent::FindInteractable()
{
    APlayerController* PlayerController = GetOwner<APlayerController>();
    if (!PlayerController) return;

    APawn* PlayerPawn = PlayerController->GetPawn();
    if (!PlayerPawn) return;

    UCameraComponent* PlayerCamera = PlayerPawn->FindComponentByClass<UCameraComponent>();
    if (!PlayerCamera) return;

    // Perform a line trace to find interactables
    FVector Start = PlayerCamera->GetComponentLocation();
    FVector End = Start + (PlayerCamera->GetForwardVector() * InteractDistance);

    FHitResult HitResult;
    FCollisionQueryParams QueryParams;
    QueryParams.AddIgnoredActor(PlayerPawn);

    if (GetWorld()->LineTraceSingleByChannel(HitResult, Start, End, ECC_Visibility, QueryParams))
    {
        AInteractableActor* Interactable = Cast<AInteractableActor>(HitResult.GetActor());
        if (Interactable)
        {
            if (Interactable != CurrentInteractable)
            {
                ClearInteractable();
                CurrentInteractable = Interactable;
                CurrentInteractable->OnLookAt();
            }
        }
        else
        {
            ClearInteractable();
        }
    }
    else
    {
        ClearInteractable();
    }
}

void UPlayerInteractComponent::ClearInteractable()
{
    if (CurrentInteractable)
    {
        CurrentInteractable->RestoreMaterials();
        CurrentInteractable = nullptr;
    }
}

{
    bHackingSuccess = bSuccess;
    UE_LOG(LogTemp, Warning, TEXT("Hacking game completed with %s"), bSuccess ? TEXT("success") : TEXT("failure"));

    if (bSuccess)
    {
        // Unlock lore or trigger a positive event
        ATerminalSystem* TerminalSystem = FindObjectOfClass<ATerminalSystem>(GetWorld());
        if (TerminalSystem)
        {
            TerminalSystem->DisplayText("> FILE UNLOCKED: //N3T/ARCHIVES/INCIDENT_3026\n> CONTENT: The bugmatter was a test. We failed.", false);
        }
    }
    else
    {
        // Trigger a negative event (e.g., security alert)
        ATerminalSystem* TerminalSystem = FindObjectOfClass<ATerminalSystem>(GetWorld());
        if (TerminalSystem)
        {
            TerminalSystem->DisplayText("> ACCESS DENIED. ALERT TRIGGERED.", true);
        }
    }
}

#include "HackingGame.h"

AHackingGame::AHackingGame()
{
    PrimaryActorTick.bCanEverTick = false;
}

void AHackingGame::BeginPlay()
{
    Super::BeginPlay();
}

void AHackingGame::StartGame()
{
    bGameActive = true;
    CurrentPassword = "";
    RemainingAttempts = MaxAttempts;
    OnHackingGameCompleted.Clear();
}

void AHackingGame::SubmitPassword(const FString& Password)
{
    if (!bGameActive) return;

    CurrentPassword = Password;
    RemainingAttempts--;

    if (Password == CorrectPassword)
    {
        // Success
        bGameActive = false;
        OnHackingGameCompleted.Broadcast(true);
    }
    else if (RemainingAttempts <= 0)
    {
        // Failure
        bGameActive = false;
        OnHackingGameCompleted.Broadcast(false);
    }
    else
    {
        // Wrong password, try again
        UE_LOG(LogTemp, Warning, TEXT("Wrong password. Attempts left: %d"), RemainingAttempts);
    }
}

void UHackingGameWidget::NativeConstruct()

    // Bind button click events
    if (ButtonO) ButtonO->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("O"));
    if (ButtonV) ButtonV->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("V"));
    if (ButtonE) ButtonE->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("E"));
    if (ButtonR) ButtonR->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("R"));
    if (ButtonI) ButtonI->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("I"));
    if (ButtonD) ButtonD->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("D"));
    if (ButtonL) ButtonL->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("L"));
    if (ButtonEnter) ButtonEnter->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("ENTER"));
    if (ButtonDelete) ButtonDelete->OnClicked.AddDynamic(this, &UHackingGameWidget::OnButtonClicked, FString("DELETE"));

    // Find the HackingGame actor
    HackingGame = FindObjectOfClass<AHackingGame>(GetWorld());
    if (HackingGame)
    {
        HackingGame->OnHackingGameCompleted.AddDynamic(this, &UHackingGameWidget::ShowResult);
    }
}

void UHackingGameWidget::StartHackingGame()
{
    if (HackingGame)
    {
        HackingGame->StartGame();
        UpdatePasswordDisplay("");
        UpdateAttemptsDisplay(HackingGame->GetRemainingAttempts());
        FeedbackText->SetText(FText::FromString(""));
        SuccessPanel->SetVisibility(ESlateVisibility::Collapsed);
        FailurePanel->SetVisibility(ESlateVisibility::Collapsed);
    }
}

void UHackingGameWidget::OnButtonClicked(FString Letter)
{
    if (!HackingGame || !HackingGame->IsGameActive()) return;

    if (Letter == "ENTER")
    {
        SubmitPassword(PasswordText->GetText().ToString());
    }
    else if (Letter == "DELETE")
    {
        FString CurrentPassword = PasswordText->GetText().ToString();
        if (CurrentPassword.Len() > 0)
        {
            CurrentPassword.RemoveAt(CurrentPassword.Len() - 1);
            UpdatePasswordDisplay(CurrentPassword);
        }
    }
    else
    {
        FString CurrentPassword = PasswordText->GetText().ToString();
        if (CurrentPassword.Len() < HackingGame->GetCorrectPassword().Len())
        {
            CurrentPassword += Letter;
            UpdatePasswordDisplay(CurrentPassword);
        }
    }
}

void UHackingGameWidget::SubmitPassword(const FString& Password)
{
    if (HackingGame)
    {
        HackingGame->SubmitPassword(Password);
    }
}

void UHackingGameWidget::UpdatePasswordDisplay(const FString& Password)
{
    if (PasswordText)
    {
        PasswordText->SetText(FText::FromString(Password));
    }
}

void UHackingGameWidget::UpdateAttemptsDisplay(int32 AttemptsLeft)
{
    if (AttemptsText)
    {
        AttemptsText->SetText(FText::FromString(FString::Printf(TEXT("ATTEMPTS LEFT: %d"), AttemptsLeft)));
    }
}

void UHackingGameWidget::ShowResult(bool bSuccess)
{
    if (bSuccess)
    {
        FeedbackText->SetText(FText::FromString("ACCESS GRANTED"));
        SuccessPanel->SetVisibility(ESlateVisibility::Visible);
        FailurePanel->SetVisibility(ESlateVisibility::Collapsed);
    }
    else
    {
        FeedbackText->SetText(FText::FromString("ACCESS DENIED"));
        SuccessPanel->SetVisibility(ESlateVisibility::Collapsed);
        FailurePanel->SetVisibility(ESlateVisibility::Visible);
    }
}
