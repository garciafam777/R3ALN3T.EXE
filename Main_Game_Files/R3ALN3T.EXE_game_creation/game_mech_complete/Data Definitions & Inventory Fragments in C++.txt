1. Data Definitions & Inventory Framework
First, we establish a globally accessible Enum and Structure layer to define our Bugmatter types and career paths. This ensures that both Claude's gameplay classes and Gemini's architecture blocks read from the same exact data structure.

Source/Public/Core/Types/GameTypes.h
C++
#pragma once

#include "CoreMinimal.h"
#include "Engine/UserDefinedStruct.h"
#include "GameTypes.generated.h"

// The four core Bugmatter states physicalizing in 3026
UENUM(BlueprintType)
enum class EBugmatterType : uint8
{
    SootMatter    UMETA(DisplayName = "Soot-Matter (Logic Overflow)"),
    CrustMatter   UMETA(DisplayName = "Crust-Matter (Industrial Slag)"),
    VesselMatter  UMETA(DisplayName = "Vessel-Matter (Bio-Digital Cultured)"),
    SludgeMatter  UMETA(DisplayName = "Sludge-Matter (Atmospheric Runoff)")
};

// The four career paths driving the sector economies
UENUM(BlueprintType)
enum class EPlayerCareer : uint8
{
    SublimationTuner  UMETA(DisplayName = "Sublimation Tuner"),
    LogicSplicer      UMETA(DisplayName = "Logic Splicer"),
    NaviCultivator    UMETA(DisplayName = "Navi-Cultivator"),
    GridScavenger     UMETA(DisplayName = "Grid Scavenger")
};

USTRUCT(BlueprintType)
struct FBugmatterItemData
{
    GENERATED_BODY()

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Inventory")
    EBugmatterType MatterType;

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Inventory")
    float Volatility; // 0.0 (Stable) to 1.0 (Critical Glitch Risk)

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Inventory")
    float Purity; // Affects processing yields and craft success rates

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Inventory")
    int32 Quantity;

    FBugmatterItemData()
        : MatterType(EBugmatterType::SludgeMatter), Volatility(0.1f), Purity(0.5f), Quantity(1) {}
};
2. The Career Progression Component
This actor component attaches directly to the local player character. It tracks experience nodes, computes career-specific perk thresholds, and checks lock status for synthesis activities.

C++
#include "Gameplay/Careers/CareerComponent.h"

UCareerComponent::UCareerComponent()
{
    PrimaryComponentTick.bCanEverTick = false;
    BaseXPRequirement = 1000.0f;
    XPExponentialMultiplier = 1.5f;

    // Initialize the mapping structures
    CareerLevels.Add(EPlayerCareer::SublimationTuner, 1);
    CareerLevels.Add(EPlayerCareer::LogicSplicer, 1);
    CareerLevels.Add(EPlayerCareer::NaviCultivator, 1);
    CareerLevels.Add(EPlayerCareer::GridScavenger, 1);

    CareerXP.Add(EPlayerCareer::SublimationTuner, 0.0f);
    CareerXP.Add(EPlayerCareer::LogicSplicer, 0.0f);
    CareerXP.Add(EPlayerCareer::NaviCultivator, 0.0f);
    CareerXP.Add(EPlayerCareer::GridScavenger, 0.0f);
}

void UCareerComponent::BeginPlay()
{
    Super::BeginPlay();
}

void UCareerComponent::AddXP(EPlayerCareer Career, float Amount)
{
    if (!CareerXP.Contains(Career) || !CareerLevels.Contains(Career)) return;

    CareerXP[Career] += Amount;
    
    // Formula calculation for next level threshold
    float NextLevelThreshold = BaseXPRequirement * FMath::Pow(XPExponentialMultiplier, CareerLevels[Career] - 1);

    while (CareerXP[Career] >= NextLevelThreshold)
    {
        CareerXP[Career] -= NextLevelThreshold;
        CareerLevels[Career]++;
        UE_LOG(LogTemp, Log, TEXT("Career %d leveled up to %d!"), (int32)Career, CareerLevels[Career]);
        
        NextLevelThreshold = BaseXPRequirement * FMath::Pow(XPExponentialMultiplier, CareerLevels[Career] - 1);
    }
}

int32 UCareerComponent::GetCareerLevel(EPlayerCareer Career) const
{
    return CareerLevels.Contains(Career) ? CareerLevels[Career] : 0;
}

float UCareerComponent::GetSynthesisYieldModifier(EPlayerCareer Career) const
{
    int32 CurrentLevel = GetCareerLevel(Career);
    // Returns efficiency scaling factor (e.g., Level 10 gives +20% output)
    return 1.0f + (CurrentLevel * 0.02f);
}

bool UCareerComponent::CanProcessMatterType(EPlayerCareer Career, EBugmatterType MatterType) const
{
    int32 Level = GetCareerLevel(Career);

    // Cross-verify career unlock configurations
    switch (Career)
    {
        case EPlayerCareer::LogicSplicer:
            return (MatterType == EBugmatterType::SootMatter && Level >= 1);
        case EPlayerCareer::SublimationTuner:
            return (MatterType == EBugmatterType::CrustMatter && Level >= 1);
        case EPlayerCareer::NaviCultivator:
            return (MatterType == EBugmatterType::VesselMatter && Level >= 1);
        case EPlayerCareer::GridScavenger:
            return true; // Street operators can process or refine low-tier sludge natively
        default:
            return false;
    }
}
3. The Local System Hardware Manager (Synthesis Node)
This architecture class interfaces directly with localized server hardware loops. It manages the risk metrics, processing heat limits, and structural output generation when executing synthesis.

Source/Public/Core/Memory/SynthesisNode.h

#include "CoreMinimal.h"

#include "Core/Types/GameTypes.h"

UCLASS()

{
    GENERATED_BODY()

private:
    UPROPERTY(EditAnywhere, Category = "Hardware Configuration")
    float CoreThermalLimit;

    UPROPERTY(VisibleAnywhere, Category = "Telemetry")
    float CurrentCoreTemperature;

    UPROPERTY(VisibleAnywhere, Category = "Telemetry")
    bool bIsSynthesisActive;

    // Processes raw data or materials into stable physical assets
    UFUNCTION(BlueprintCallable, Category = "Synthesis Engine")
    bool TriggerMemoryOverflowSynthesis(AActor* InstigatorCharacter, FBugmatterItemData RawInput, FBugmatterItemData& OutCompiledMatter);

    UFUNCTION(BlueprintPure, Category = "Telemetry")
    float GetThermalPercentage() const { return CurrentCoreTemperature / CoreThermalLimit; }
};

ASynthesisNode::ASynthesisNode()
{
    PrimaryActorTick.bCanEverTick = false;
    CoreThermalLimit = 120.0f; // Critical Celsius ceiling for unshielded local boards
    CurrentCoreTemperature = 35.0f;
    bIsSynthesisActive = false;
}

void ASynthesisNode::BeginPlay()
{
    Super::BeginPlay();
}

bool ASynthesisNode::TriggerMemoryOverflowSynthesis(AActor* InstigatorCharacter, FBugmatterItemData RawInput, FBugmatterItemData& OutCompiledMatter)
{
    if (!InstigatorCharacter || bIsSynthesisActive) return false;

    UCareerComponent* CareerComp = InstigatorCharacter->FindComponentByClass<UCareerComponent>();
    if (!CareerComp) return false;

    // Rule protection matrix
    if (!CareerComp->CanProcessMatterType(EPlayerCareer::LogicSplicer, RawInput.MatterType))
    {
        UE_LOG(LogTemp, Warning, TEXT("[Synthesis] Execution blocked: Instigator lacks structural clearance."));
        return false;
    }

    bIsSynthesisActive = true;
    
    // Calculate heat delta based on input volatility and purity vectors
    float HeatGeneration = (RawInput.Volatility * 45.0f) / (RawInput.Purity + 0.1f);
    CurrentCoreTemperature += HeatGeneration;

    if (CurrentCoreTemperature >= CoreThermalLimit)
    {
        UE_LOG(LogTemp, Error, TEXT("[Hardware] Thermal runaway event triggered. Local database dump corrupted."));
        CurrentCoreTemperature = CoreThermalLimit;
        bIsSynthesisActive = false;
        return false; // Hardware failure state
    }

    // Process output variables
    OutCompiledMatter.MatterType = RawInput.MatterType;
    OutCompiledMatter.Purity = FMath::Clamp(RawInput.Purity * 1.15f, 0.0f, 1.0f);
    OutCompiledMatter.Volatility = FMath::Clamp(RawInput.Volatility * 0.8f, 0.0f, 1.0f);
    OutCompiledMatter.Quantity = FMath::FloorToInt(RawInput.Quantity * CareerComp->GetSynthesisYieldModifier(EPlayerCareer::LogicSplicer));

    // Award career progression XP metrics
    CareerComp->AddXP(EPlayerCareer::LogicSplicer, 250.0f * RawInput.Purity);

    bIsSynthesisActive = false;
    return true;
}
