The 3026 Aging FormulaThe aging calculation must perfectly mirror your specific scale: For every 1 human year in real life, the NetP character ages exactly $0.10\%$ of a real-life year.To convert this into standard gameplay metrics:$1.0$ Real-Life Year = $365$ Days.$0.10\%$ of $365$ Days = $0.365$ Days ($8.76$ Hours) of baseline character aging per real-life calendar year.Trauma events (accidents) apply a flat Aging Penalty Vector, artificially jumping the internal clock forward due to hardware strain.2. 3. Injury & Chronological Progression LogicSource/Private/Gameplay/Systems/NetPLifeComponent.cppC++#include "Gameplay/Systems/NetPLifeComponent.h"
#include "Components/SkeletalMeshComponent.h"
#include "GameFramework/Actor.h"

UNetPLifeComponent::UNetPLifeComponent()
{
    PrimaryActorTick.bCanEverTick = false;
    CurrentCharacterAge = 21.000f; // Starting baseline adult age
    MaxNaturalAgeCeiling = 120.000f; // Ultimate terminal lifecycle point
    HardwareFragmentation = 0.0f;
}

void UNetPLifeComponent::BeginPlay()
{
    Super::BeginPlay();
}

void UNetPLifeComponent::ProgressNaturalAge(float RealLifeYearsPassed)
{
    // 1 Human Year = 0.10% aging acceleration factor (0.001 fractional years)
    float AgeDelta = RealLifeYearsPassed * 0.001f;
    CurrentCharacterAge += AgeDelta;

    ApplyVisualAgingToMesh();
}

void UNetPLifeComponent::ProcessAccidentEvent(EAccidentType Accident, float Severity)
{
    // Clamp incoming impact scalar parameters
    Severity = FMath::Clamp(Severity, 0.1f, 1.0f);
    
    // Accidents permanently degrade hardware, forcing a logic age-jump
    float AgePenalty = 0.0f;
    float FractureDelta = 0.0f;

    switch (Accident)
    {
        case EAccidentType::VehicleCrash:
            AgePenalty = 0.25f * Severity;    // High physical stress
            FractureDelta = 0.40f * Severity;
            break;
        case EAccidentType::MiningCaveIn:
            AgePenalty = 0.50f * Severity;    // Major crushing trauma
            FractureDelta = 0.60f * Severity;
            break;
        case EAccidentType::GridOverload:
            AgePenalty = 0.15f * Severity;    // Direct logic scarring
            FractureDelta = 0.75f * Severity;
            break;
    }

    CurrentCharacterAge += AgePenalty;
    HardwareFragmentation = FMath::Clamp(HardwareFragmentation + FractureDelta, 0.0f, 1.0f);

    // Force transition hook directly to the Hospital sector map
    UE_LOG(LogTemp, Warning, TEXT("[TRAUMA] Critical failure. NetP forced into Reassembly. Age Penalty Applied: +%f"), AgePenalty);
    
    ApplyVisualAgingToMesh();
}

void UNetPLifeComponent::RestInHospitalFacility(float ServiceCost, float RepairDurationRealHours)
{
    // Going to the hospital repairs physical layout degradation, but cannot reverse age vectors
    HardwareFragmentation = FMath::Max(0.0f, HardwareFragmentation - (RepairDurationRealHours * 0.1f));
    
    // Natural age progresses marginally during long medical operations
    ProgressNaturalAge(RepairDurationRealHours / 8760.0f); // Hours converted to fractional years
}

void UNetPLifeComponent::ApplyVisualAgingToMesh()
{
    AActor* Owner = GetOwner();
    if (!Owner) return;

    USkeletalMeshComponent* Mesh = Owner->FindComponentByClass<USkeletalMeshComponent>();
    if (Mesh)
    {
        UMaterialInstanceDynamic* MID = Cast<UMaterialInstanceDynamic>(Mesh->GetMaterial(0));
        if (MID)
        {
            // Normalize current age against standard lifespan for structural shader parameters
            float AgeNormalized = (CurrentCharacterAge - 21.0f) / (MaxNaturalAgeCeiling - 21.0f);
            
            // Drives skin roughness adjustments, loss of synthetic melanin gleam, or chrome tarnish
            MID->SetScalarParameterValue(TEXT("ChronologicalDegradation"), FMath::Clamp(AgeNormalized, 0.0f, 1.0f));
            MID->SetScalarParameterValue(TEXT("HardwareFragmentationScars"), HardwareFragmentation);
        }
    }
}

When high, certain emissive neon patterns on the NetP will flicker, short-circuit, or show cracked, raw circuit paths beneath the outer plating.
