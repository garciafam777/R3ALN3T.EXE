UENUM(BlueprintType)
enum class ETileSide : uint8

    PlayerSide   UMETA(DisplayName = "Player Territory (Purple)"),
    EnemySide    UMETA(DisplayName = "Enemy Territory (Teal/Grey)")

USTRUCT(BlueprintType)
struct FGridCoordinate
{
    GENERATED_BODY()

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Battle Grid")
    int32 X; // Range 0 to 3

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Battle Grid")
    int32 Y; // Range 0 to 3

    FGridCoordinate() : X(0), Y(0) {}
    FGridCoordinate(int32 InX, int32 InY) : X(InX), Y(InY) {}

    bool operator==(const FGridCoordinate& Other) const
    {
        return X == Other.X && Y == Other.Y;
    }
};

Source/Private/Gameplay/Battle/BattleGridManager.cppC++#include "Gameplay/Battle/BattleGridManager.h"

ABattleGridManager::ABattleGridManager()
{
    PrimaryActorTick.bCanEverTick = true;
    CustomGaugePercentage = 0.0f;
    CustomGaugeFillRate = 20.0f; // Takes 5 seconds to hit 100% Custom state
}

void ABattleGridManager::BeginPlay()
{
    Super::BeginPlay();
}

void ABattleGridManager::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);

    // Charge the interface gauge over real-time frames
    if (CustomGaugePercentage < 100.0f)
    {
        CustomGaugePercentage = FMath::Clamp(CustomGaugePercentage + (CustomGaugeFillRate * DeltaTime), 0.0f, 100.0f);
    }
}

bool ABattleGridManager::RequestTileMove(ETileSide CurrentSide, FGridCoordinate CurrentPos, FGridCoordinate TargetPos, FVector& OutWorldLocation)
{
    // Bound checking for the 4x4 sector array
    if (TargetPos.X < 0 || TargetPos.X >= MaxCols || TargetPos.Y < 0 || TargetPos.Y >= MaxRows)
    {
        return false;
    }

    // Unless you use your MCADMIN privileges to break the field rules, 
    // baseline NetP units cannot hop onto the opposing faction's side natively.
    // X dimensions are local to the respective 4x4 fields.
    
    // Grid spacing allocation calculations (Assuming 200 units per holographic panel tile)
    float TileSpacing = 200.0f;
    float SideOffset = (CurrentSide == ETileSide::EnemySide) ? 900.0f : 0.0f; // Distinct split gap

    OutWorldLocation = FVector(TargetPos.X * TileSpacing + SideOffset, TargetPos.Y * TileSpacing, 0.0f);
    return true;
}
