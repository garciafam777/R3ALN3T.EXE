UENUM(BlueprintType)
enum class EChipClass : uint8
{
    Standard   UMETA(DisplayName = "Standard Code"),
    Mega       UMETA(DisplayName = "Mega Directive"),

USTRUCT(BlueprintType)
struct FDynamicBattleChipData
{
    GENERATED_BODY()

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Matrix")
    FString ChipName;

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Matrix")
    EChipClass ChipTier;

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Matrix")
    int32 BaseDamage;

    // Mutated modifiers rolled dynamically when drawn into the active hand
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Mutation")
    float DamageMultiplier;

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Mutation")
    float InstabilityGlitchChance; // Higher chance triggers backfire/reduced accuracy

    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "Chip Mutation")
    FString CustomModifierTag; // e.g., "PIERCE", "STUN", "REGEN"

    FDynamicBattleChipData()
        : ChipName(TEXT("Buster_Bolt")), ChipTier(EChipClass::Standard), BaseDamage(40)
        , DamageMultiplier(1.0f), InstabilityGlitchChance(0.0f), CustomModifierTag(TEXT("NONE")) {}
};

3. Dynamic Roll & Backdoor Injection LogicSource/Private/Gameplay/Battle/BattleFolderManager.cppC++#include "Gameplay/Battle/BattleFolderManager.h"
#include "Kismet/KismetMathLibrary.h"

UBattleFolderManager::UBattleFolderManager()
{
    PrimaryActorTick.bCanEverTick = false;
}

void UBattleFolderManager::BeginPlay()
{
    Super::BeginPlay();

    // Mocking a baseline starter deck setup inside the master folder array
    for (int32 i = 0; i < 30; i++)
    {
        FDynamicBattleChipData DefaultChip;
        if (i % 10 == 0) {
            DefaultChip.ChipName = TEXT("Sword_Matrix");
            DefaultChip.BaseDamage = 80;
        } else if (i == 29) {
            // A hidden placeholder card that scales with admin privileges
            DefaultChip.ChipName = TEXT("MCADMIN_Purge_Protocol");
            DefaultChip.ChipTier = EChipClass::Admin;
            DefaultChip.BaseDamage = 9999;
        }
        MasterFolder.Add(DefaultChip);
    }
}

void UBattleFolderManager::ApplyDataMutation(FDynamicBattleChipData& TargetChip)
{
    // If you are piloting one of your disguised NetP hulls, bypass negative mutations completely
    if (TargetChip.ChipTier == EChipClass::Admin)
    {
        TargetChip.DamageMultiplier = 2.5f;
        TargetChip.InstabilityGlitchChance = 0.0f;
        TargetChip.CustomModifierTag = TEXT("DIRECT_CORE_ERASE");
        return;
    }

    // Standard Player/NetP calculations: roll variance based on background algorithm noise
    float Roll = UKismetMathLibrary::RandomFloatInRange(0.0f, 1.0f);
    
    if (Roll > 0.85f) // 15% Chance of a positive data surge
    {
        TargetChip.DamageMultiplier = UKismetMathLibrary::RandomFloatInRange(1.1f, 1.35f);
        TargetChip.CustomModifierTag = TEXT("OVERCHARGED");
    }
    else if (Roll < 0.10f) // 10% Chance of drawing corrupted code fragment
    {
        TargetChip.DamageMultiplier = UKismetMathLibrary::RandomFloatInRange(0.7f, 0.9f);
        TargetChip.InstabilityGlitchChance = UKismetMathLibrary::RandomFloatInRange(0.15f, 0.4f);
        TargetChip.CustomModifierTag = TEXT("LOGIC_BLEED");
    }
    else
    {
        TargetChip.DamageMultiplier = 1.0f;
        TargetChip.InstabilityGlitchChance = 0.05f; // Baseline global interference
        TargetChip.CustomModifierTag = TEXT("STABLE");
    }
}

void UBattleFolderManager::DrawNewCombatPhaseHand(int32 HandSize)
{
    CurrentActiveHand.Empty();
    
    if (MasterFolder.Num() == 0) return;

    // Draw up to max hand size, shuffling indexes dynamically
    for (int32 i = 0; i < HandSize; i++)
    {
        int32 RandomIndex = UKismetMathLibrary::RandomInteger(MasterFolder.Num());
        FDynamicBattleChipData PickedChip = MasterFolder[RandomIndex];
        
        // Mutate the stats of the chip instances right as they hit your glove queue
        ApplyDataMutation(PickedChip);
        
        CurrentActiveHand.Add(PickedChip);
    }
    
    UE_LOG(LogTemp, Log, TEXT("[Combat Hub] New customized Battle Chips compiled into the active hand array."));
}

bool UBattleFolderManager::ExecuteChipActivation(int32 HandIndex, FDynamicBattleChipData& OutUsedChipData)
{
    if (!CurrentActiveHand.IsValidIndex(HandIndex)) return false;

    OutUsedChipData = CurrentActiveHand[HandIndex];
    
    // Remove from active selection hand once processed by the grid execution pipeline
    CurrentActiveHand.RemoveAt(HandIndex);
    return true;
}
