SQL-- =============================================
-- R3ALN3T.EXE Dialogue Database Schema
-- Version: 1.0 (2026-06-29)
-- =============================================

PRAGMA foreign_keys = ON;

-- =============================================
-- 1. CORE PROFILES
-- =============================================

-- NetP Personalities (1000+ entries)
CREATE TABLE netp_profiles (
    id TEXT PRIMARY KEY,                    -- e.g., "lex_001"
    name TEXT NOT NULL,
    display_name TEXT,
    gender TEXT,
    age INTEGER,
    personality_json TEXT NOT NULL,         -- Full traits, speech_style, voice_preset, etc.
    created_at INTEGER DEFAULT (unixepoch()),
    updated_at INTEGER DEFAULT (unixepoch())
);

-- Player Relationship Data (per NetP)
CREATE TABLE player_relationships (
    netp_id TEXT NOT NULL,
    player_id TEXT NOT NULL DEFAULT 'main_player',
    relationship_level INTEGER DEFAULT 0,   -- 0-100
    trust_score INTEGER DEFAULT 50,
    last_interaction INTEGER,
    flags TEXT,                             -- JSON for special states
    PRIMARY KEY (netp_id, player_id),
    FOREIGN KEY (netp_id) REFERENCES netp_profiles(id)
);

-- =============================================
-- 2. DIALOGUE TEMPLATES & CONTENT
-- =============================================

-- Base Templates (Tens of thousands)
CREATE TABLE dialogue_templates (
    id TEXT PRIMARY KEY,
    category TEXT NOT NULL,                 -- "ambient_greeting", "quest_complete", "home_visit", etc.
    context_tags TEXT,                      -- comma-separated or JSON
    base_text TEXT NOT NULL,
    emotion_tag TEXT,                       -- happy, serious, sarcastic, etc.
    speaker_role TEXT,                      -- "NetP", "Player", "Narrator"
    priority INTEGER DEFAULT 50,
    min_relationship INTEGER DEFAULT 0,
    max_relationship INTEGER DEFAULT 100
);

-- Story & Cutscene Scripts
CREATE TABLE story_scenes (
    scene_id TEXT PRIMARY KEY,
    title TEXT,
    is_cutscene BOOLEAN DEFAULT 0,
    lines_json TEXT NOT NULL,               -- Array of {speaker, text, choices, next_scene}
    required_flags TEXT,                    -- JSON array of story flags
    reward_flags TEXT                       -- Flags set after completion
);

-- =============================================
-- 3. CACHING & GENERATED CONTENT
-- =============================================

-- Generated Dialogue Cache (Millions of entries)
CREATE TABLE dialogue_cache (
    cache_hash TEXT PRIMARY KEY,            -- SHA256(netp_id + template_id + context + relationship)
    netp_id TEXT NOT NULL,
    template_id TEXT,
    generated_text TEXT NOT NULL,
    emotion_tag TEXT,
    timestamp INTEGER DEFAULT (unixepoch()),
    used_count INTEGER DEFAULT 1,
    expires_at INTEGER,                     -- For temporary context lines
    FOREIGN KEY (netp_id) REFERENCES netp_profiles(id)
);

-- =============================================
-- 4. INDEXES FOR PERFORMANCE
-- =============================================

CREATE INDEX idx_netp_name ON netp_profiles(name);
CREATE INDEX idx_relationship_netp ON player_relationships(netp_id);
CREATE INDEX idx_template_category ON dialogue_templates(category);
CREATE INDEX idx_cache_netp ON dialogue_cache(netp_id);
CREATE INDEX idx_cache_hash ON dialogue_cache(cache_hash);
CREATE INDEX idx_cache_timestamp ON dialogue_cache(timestamp);

-- =============================================
-- 5. VIEWS & UTILITY
-- =============================================

-- Quick view of active relationships
CREATE VIEW v_active_relationships AS
SELECT 
    p.name,
    r.relationship_level,
    r.trust_score
FROM player_relationships r
JOIN netp_profiles p ON r.netp_id = p.id;

-- Recent cache usage
CREATE VIEW v_cache_stats AS
SELECT 
    COUNT(*) as total_cached,
    AVG(used_count) as avg_uses
FROM dialogue_cache;
