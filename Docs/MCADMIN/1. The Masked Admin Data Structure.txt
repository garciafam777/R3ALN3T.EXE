2. Multi-Vessel Handoff Implementation
This logic handles the seamless swapping of your characters on the isometric map. When you trigger a swap, the engine caches your physical position, unpossesses the current NetP chassis, applies your masked identity to the new shell, and updates your appearance instantly on the GPU.

Source/Private/Core/Admin/MCAdminMaskComponent.cpp
C++
#include "Core/Admin/MCAdminMaskComponent.h"
#include "Gameplay/Character/NetNaviCharacter.h" // Linked to NetP file
#include "GameFramework/PlayerController.h"

UMCAdminMaskComponent::UMCAdminMaskComponent()
{
    PrimaryActorTick.bCanEverTick = false;
    CurrentActiveVesselIndex = 0;
    bIsAdminOverrideActive = true;

    // Initialize your 3 personal NetP vessels with distinct spoofed profiles
    for (int32 i = 0; i < 3; i++)
    {
        FNetPVesselChassis NewChassis;
        NewChassis.VesselID = i;
        NewChassis.MaskedUsername = FString::Printf(TEXT("Operator_Alpha_%d"), i);
        NewChassis.SpoofedAge = 21.0f + (i * 4.5f); // Varied age layers to dodge suspicion
        MaskedNetPSlots.Add(NewChassis);
    }
}

bool UMCAdminMaskComponent::SwapActiveNetPVessel(int32 TargetSlotIndex)
{
    if (!MaskedNetPSlots.IsValidIndex(TargetSlotIndex) || TargetSlotIndex == CurrentActiveVesselIndex)
    {
        return false;
    }

    AActor* OwnerActor = GetOwner();
    APlayerController* PC = Cast<APlayerController>(OwnerActor);
    if (!PC) return false;

    APawn* CurrentPawn = PC->GetPawn();
    if (!CurrentPawn) return false;

    FVector CurrentLocation = CurrentPawn->GetActorLocation();
    FRotator CurrentRotation = CurrentPawn->GetActorRotation();

    // Cache the old vessel data before termination
    MaskedNetPSlots[CurrentActiveVesselIndex].CustomAppearance = FCharacterAppearanceData(); // Stores runtime updates

    // Destroy current physical shell in the level
    CurrentPawn->Destroy();

    // Spawn the next 4K NetP chassis from your reserved administrative slots
    FActorSpawnParameters SpawnParams;
    SpawnParams.SpawnCollisionHandlingOverride = ESpawnActorCollisionHandlingMethod::AlwaysSpawn;
    
    // Assumes your core NetP Character class is configured for runtime instantiation
    ANetNaviCharacter* NewNetP = GetWorld()->SpawnActor<ANetNaviCharacter>(ANetNaviCharacter::StaticClass(), CurrentLocation, CurrentRotation, SpawnParams);
    
    if (NewNetP)
    {
        CurrentActiveVesselIndex = TargetSlotIndex;
        
        // Possess the new shell disguised as a normal player
        PC->Possess(NewNetP);
        
        // Push the dynamic material variables to the GPU for this specific vessel profile
        NewNetP->UpdateAppearance(MaskedNetPSlots[CurrentActiveVesselIndex].CustomAppearance);
        
        UE_LOG(LogTemp, Log, TEXT("[MCADMIN] Consciousness transferred cleanly to NetP Slot %d. Identity masked."), CurrentActiveVesselIndex);
        return true;
    }

    return false;
}
